1"""Open3D visualization utilities for event camera data. 

2 

3This module provides functions to visualize event streams in 3D (x, y, time) 

4using Open3D. 

5""" 

6import open3d as o3d 

7import numpy as np 

8from matplotlib import pyplot as plt 

9 

10def o3d_draw_events(events: np.ndarray) -> None: 

11 """Visualizes events using Open3D. 

12 

13 Parameters 

14 ---------- 

15 events : np.ndarray 

16 Array of events with fields 'x', 'y', 't', and 'p'. 

17 'x' and 'y' are the pixel coordinates, 't' is the timestamp, and 'p' is the polarity. 

18 

19 Returns 

20 ------- 

21 None 

22 

23 """ 

24 if len(events) == 0: 

25 return 

26 

27 # Draw X, Y, T as a pointcloud  

28 pdc = o3d.geometry.PointCloud() 

29 

30 time_diff = np.max(events['t']) - np.min(events['t']) 

31 # Normalize time to range 0..3000 

32 norm_time = (events['t'] - np.min(events['t'])) / time_diff * 3000 

33 

34 pdc.points = o3d.utility.Vector3dVector(np.column_stack((events['x'], events['y'], norm_time))) 

35 

36 # Create a color map based on the 'p' 

37 # p can be either 0 or 1, we use it to color the points witht eh Spectral colormap 

38 colors = plt.cm.Spectral(events['p'].astype(np.float32))[:, :3] # Use only RGB channels 

39 

40 pdc.colors = o3d.utility.Vector3dVector(colors) 

41 

42 o3d.visualization.draw_geometries([pdc])
