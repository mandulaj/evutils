1"""Regression tests for correctness bugs found in review (July 2026). 

2 

3Each test pins one confirmed bug so the fix stays in place: 

4 

51. Double-seek timestamp corruption -- a second ``seek()`` must not inherit the 

6 first seek's TIME_HIGH wrap correction (``_evt.py`` ``seek``). 

72. Dense kernels scrambling coordinates -- structured arrays whose fields are

9 (``jit.py`` ``lazy_njit_unwrapped_events``). 

103. EVT3 timestamps overflowing at ~71.6 min -- both the scalar and the

25# Bug 1: double seek must not inherit the previous seek's wrap correction 

26# --------------------------------------------------------------------------- # 

27def _ramp(n=20_000, dt=1_000): 

28 ev = np.zeros(n, dtype=Event_dtype) 

29 ev["t"] = np.arange(n, dtype=np.int64) * dt 

30 ev["x"] = np.arange(n) % 1280 

31 ev["y"] = np.arange(n) % 720 

32 ev["p"] = np.arange(n) % 2 

33 return ev

36@pytest.mark.parametrize("fmt", ["evt3", "evt2", "evt21"]) 

37def test_double_seek_across_wrap_then_drain(tmp_path, fmt): 

38 """First seek lands past the EVT3 TIME_HIGH wrap (2**24 µs); the second seek 

39 to a pre-wrap target must then read a contiguous, correctly-timestamped tail. 

40 

41 Before the fix the stale ``_seek_correction`` from the first seek turned the 

42 second seek's correction into ``W2 - W1``; the staged boundary chunk came out 

43 right but every chunk after it was offset by ``-W1``. 

44 """ 

45 ev = _ramp() 

46 p = tmp_path / f"ramp.{fmt}.raw" 

47 with EventWriter(str(p), format=fmt) as w: 

48 w.write(ev) 

49 

50 with EventReader(str(p), n_events=1000) as r: 

51 r.seek(t=18_000_000) # past 2**24 = 16_777_216 (EVT3 wrap) 

52 T = 2_000_000 # backward, pre-wrap 

53 exp = int(np.searchsorted(ev["t"], T)) 

54 r.seek(t=T) 

55 got = [] 

56 while True: 

57 c = r.read() 

58 if len(c) == 0: 

59 break 

60 got.append(np.asarray(c.t).copy()) 

61 

62 tail = np.concatenate(got) 

63 assert np.array_equal(tail, ev["t"][exp:])

94# Bug 3: EVT3 timestamps must survive past the uint32 (~71.6 min) ceiling 

95# --------------------------------------------------------------------------- # 

96def _evt3_word(packet_type, data): 

97 return np.uint16(((packet_type & 0xF) << 12) | (data & 0x0FFF))

100def _write_evt3_overflow_stream(path, wraps=257): 

101 """Craft a minimal EVT3 payload whose time base wraps ``wraps`` times, so the 

102 accumulator exceeds 2**32 µs, then emit one scalar and one vector event that 

103 share that timestamp. ``wraps=257`` gives ts = 257 * 2**24 + 100 > 2**32. 

104 """ 

105 TH, TL, ADDR_Y, ADDR_X, VECT_BASE_X, VECT_12, OTHERS = 0x8, 0x6, 0x0, 0x2, 0x3, 0x4, 0xE 

106 words = [] 

107 # Each (0xFFF, 0x000) TIME_HIGH pair makes the 24-bit field wrap once. 

108 for _ in range(wraps): 

109 words.append(_evt3_word(TH, 0xFFF)) 

110 words.append(_evt3_word(TH, 0x000)) 

111 words.append(_evt3_word(TL, 0x064)) # ts = wraps*2**24 + 100 

112 words.append(_evt3_word(ADDR_Y, 5)) 

113 words.append(_evt3_word(ADDR_X, 10 | (1 << 11))) # scalar: x=10, p=1 

114 words.append(_evt3_word(VECT_BASE_X, 0x100 | (1 << 11))) 

115 words.append(_evt3_word(VECT_12, 0x001)) # one vector event at base 

116 words += [_evt3_word(OTHERS, 0)] * 8 # look-ahead / tail padding 

117 payload = np.array(words, dtype=np.uint16).tobytes() 

118 

119 header = (b"% evt 3.0\n% format EVT3;height=720;width=1280\n" 

120 b"% geometry 1280x720\n% end\n") 

121 with open(path, "wb") as f: 

122 f.write(header) 

123 f.write(payload) 

124 return wraps * (1 << 24) + 100

146def _write_evt3_corrupt_stream(path): 

147 """EVT3 payload with a stray VECT_8 (a vector continuation with no preceding 

148 base) between two otherwise-valid scalar events. The decoder must skip the 

149 bad word, keeping both valid events. 

150 """ 

151 TH, TL, ADDR_Y, ADDR_X, VECT_8, OTHERS = 0x8, 0x6, 0x0, 0x2, 0x5, 0xE 

152 words = [ 

153 _evt3_word(TH, 0x000), 

154 _evt3_word(TL, 0x064), # ts = 100 

155 _evt3_word(ADDR_Y, 5), 

156 _evt3_word(ADDR_X, 10 | (1 << 11)), # valid event #1 

157 _evt3_word(VECT_8, 0x0AA), # stray VECT_8 -> corrupt 

158 _evt3_word(ADDR_X, 20 | (1 << 11)), # valid event #2 

159 ] 

160 words += [_evt3_word(OTHERS, 0)] * 8 

161 payload = np.array(words, dtype=np.uint16).tobytes() 

162 header = (b"% evt 3.0\n% format EVT3;height=720;width=1280\n" 

163 b"% geometry 1280x720\n% end\n") 

164 with open(path, "wb") as f: 

165 f.write(header) 

166 f.write(payload)

206def _evt3_trigger_words(t, tid, value): 

207 return np.array([ 

208 0x8000 | ((t >> 12) & 0xFFF), # TIME_HIGH 

209 0x6000 | (t & 0xFFF), # TIME_LOW 

210 0xA000 | ((tid & 0xF) << 8) | (value & 1), 

211 ], dtype=np.uint16)

214def test_seek_preserves_triggers_evt3(tmp_path): 

215 """A time seek with ext_trigger=True must still decode the external triggers 

216 that follow the landing point (seek x triggers path).""" 

217 ev = np.zeros(100, dtype=Event_dtype)
